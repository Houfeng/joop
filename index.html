使用指南

如何使用？
+ 下载joop
+ 将joop.js或joop-min.js放到项目中合适的位置。
+ 在相关页面用<script src='joop的url'></script>引入joop。
创建类 $class
//最简的方式创建一个类
var Animal=$class({
    name: 'animal',
    //构造函数
    $init:function(_name){
        this.name=_name;
    }
});
//为Animal创建一个子类
var Rabbit=$class({
    $base: Animal,
    name: 'aabbit'
});
使用事件 $event
var obj={};
//您可以这样创建事件
$event(obj,'eventName').bind(function(){
    //TODO: 在此处理事件
});
//还可以这样创建
$event(obj,'eventName');
obj.eventName(function(){
    //TODO: 在此处理事件
});
//可以这样调用
obj.eventName();
使用$import
//您可以引入一个js文件
$import('abc.js',function(){
    //TODO: 在此可以使用abc中定义的对象，函数等
});
//您也可以引入一组js文件
$improt(['abc.js','efg.js'],function(){
    //TODO: 在此可以使用abc及efg中定义的对象，函数等
});
定义命名空间$namespace
$namespace("demo.abc",{
//在demo.abc下定义Animal
Animal:$class({
    name: 'animal',
    //构告函数
    $init:function(_name){
        this.name=_name;
    }
})
});

使用$using

$using("demo.abc");
//然后就可以直接使用Animal了
var an=new Animal('animal');
